SELECT u.*, COUNT(f.followeremail) AS follower_count
FROM users u
LEFT JOIN followers f ON f.email = u.email
WHERE similarity(u.username, 'so') > 0.001 
   OR similarity(u.fullname, 'so') > 0.001 
   OR similarity(u.bio, 'so') > 0.001 
   OR similarity(u.email, 'so') > 0.001
GROUP BY u.username, u.fullname, u.bio, u.email,u.id  -- Add any other columns as necessary
ORDER BY similarity(u.username, 'so') DESC, follower_count DESC;


function commentItration(data){

    mainDiv.classList.add("flex", "items-center", "mt-4", "mb-4");
  
    // Create and set the image element
    const img = document.createElement("img");
    img.classList.add("rounded-full");
    img.src = item.profile_image;
    img.alt = "profile";
    img.style.height = "50px";
    img.style.width = "50px";
  
    // Create the inner div for text and actions
    const innerDiv = document.createElement("div");
    innerDiv.classList.add("ml-3", "bo");
  
    // Create and set the paragraph element for the text
    const textPara = document.createElement("p");
    textPara.style.width='80%';
    textPara.classList.add("text-white", "font-normal", "text-sm");
    textPara.textContent = `${item.fullname}   ${item.usercomment}`;
  
    // Create the actions div
    const actionsDiv = document.createElement("div");
    actionsDiv.classList.add("flex", "justify-between", "mt-4");
    actionsDiv.style.width = "100px";
  
    // Time ago span
    const timeSpan = document.createElement("span");
    timeSpan.classList.add("text-gray-400", "cursor-pointer", "font-sans", "font-semibold", "text-sm");
    timeSpan.textContent = '1d';
  
    // Likes span
    const likesSpan = document.createElement("span");
    likesSpan.classList.add("text-gray-400", "cursor-pointer", "font-sans", "font-semibold", "text-sm");
    likesSpan.textContent = '10';
  
    // Reply span
    const replySpan = document.createElement("span");
    replySpan.classList.add("text-gray-400", "cursor-pointer", "font-sans", "font-semibold", "text-sm");
    replySpan.textContent = "Reply";
  
    // Append the spans to the actions div
    actionsDiv.append(timeSpan, likesSpan, replySpan);
  
    // Append the paragraph and actions div to the inner div
    innerDiv.append(textPara, actionsDiv);
  
    // Append the image and inner div to the main div
    mainDiv.append(img, innerDiv);
  
    // Append the main div to the container
    commentSession.appendChild(mainDiv);
